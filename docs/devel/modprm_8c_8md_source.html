86 void writePayload(
87  PIOL_ExSeis* piol,
88  size_t goff,
89  size_t lnt,
90  size_t tcnt,
91  PIOL_File_ReadDirect* ifh,
92  PIOL_File_WriteDirect* ofh)
93 {
94  size_t biggest = lnt;
95  MPI_Allreduce(
96  &lnt, &biggest, 1, MPI_UNSIGNED_LONG, MPI_MAX, MPI_COMM_WORLD);
97  size_t extra =
98  biggest / tcnt - lnt / tcnt + (biggest % tcnt > 0) - (lnt % tcnt > 0);
99 
100  for (size_t i = 0U; i < lnt; i += tcnt) {
101  size_t rblock = (i + tcnt < lnt ? tcnt : lnt - i);
102  readwriteParam(piol, goff + i, rblock, ifh, ofh);
103  }
104 
105  for (size_t i = 0U; i < extra; i++) {
106  readwriteParam(piol, goff, 0, ifh, ofh);
107  }
108 }
