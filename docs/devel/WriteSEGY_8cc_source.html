212 
213 
214  // Convert from host-endianness to SEGY endianness for writing
215 
216  if (trc != TRACE_NULL && trc != nullptr) {
217  for (size_t i = 0; i < ns * sz; i++) {
218  const auto be_trc_i_bytes =
219  to_big_endian<exseis::utils::Trace_value>(trc[i]);
220 
221  unsigned char* trc_i_bytes =
222  reinterpret_cast<unsigned char*>(&trc[i]);
223 
224  std::copy(
225  std::begin(be_trc_i_bytes), std::end(be_trc_i_bytes),
226  trc_i_bytes);
227  }
228  }
229 
230 
231  if (prm == PIOL_PARAM_NULL) {
232  obj->writeDODF(offset, ns, sz, tbuf);
233  }
234  else {
235  const size_t blockSz =
236  (trc == TRACE_NULL ? SEGY_utils::getMDSz() : SEGY_utils::getDOSz(ns));
237 
238  std::vector<unsigned char> alloc(blockSz * sz);
239  unsigned char* buf = (sz ? alloc.data() : nullptr);
240 
241  SEGY_utils::insertParam(
242  sz, prm, buf, blockSz - SEGY_utils::getMDSz(), skip);
243 
244  if (trc == TRACE_NULL) {
245  obj->writeDOMD(offset, ns, sz, buf);
246  }
247  else {
248  for (size_t i = 0; i < sz; i++) {
249  std::copy(
250  &tbuf[i * SEGY_utils::getDFSz(ns)],
251  &tbuf[(i + 1) * SEGY_utils::getDFSz(ns)],
252  &buf[i * SEGY_utils::getDOSz(ns) + SEGY_utils::getMDSz()]);
253  }
254 
255  obj->writeDO(offset, ns, sz, buf);
256  }
257  }
258 
259 
260  // Convert back from SEGY endianness to native endianness, as it was
261  // before calling this routine
262 
263  if (trc != TRACE_NULL && trc != nullptr) {
264  for (size_t i = 0; i < ns * sz; i++) {
265  const exseis::utils::Trace_value be_trc_i = trc[i];
266 
267  const unsigned char* be_trc_i_bytes =
268  reinterpret_cast<const unsigned char*>(&be_trc_i);
269 
270  trc[i] = from_big_endian<exseis::utils::Trace_value>(
271  be_trc_i_bytes[0], be_trc_i_bytes[1], be_trc_i_bytes[2],
