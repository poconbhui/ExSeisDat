95 void utm2LatLong(
96  exseis::utils::Floating_point easting,
97  exseis::utils::Floating_point northing,
98  std::string utmZone,
99  exseis::utils::Floating_point& lat,
100  exseis::utils::Floating_point& lng)
101 {
102  exseis::utils::Floating_point hemi =
103  (std::toupper(utmZone.back()) == 'N' ? 1 : -1);
104  exseis::utils::Floating_point numZone = std::stof(utmZone);
105 
106  exseis::utils::Floating_point const eqRad = 6378137;
107  exseis::utils::Floating_point const polRad = 6356752;
108  exseis::utils::Floating_point const k0 = 0.9996;
109  exseis::utils::Floating_point E =
110  std::sqrt(1 - (polRad * polRad) / (eqRad * eqRad));
111  exseis::utils::Floating_point ei =
112  (1 - std::sqrt(1 - E * E)) / (1 + std::sqrt(1 - E * E));
113  exseis::utils::Floating_point x = 500000 - easting;
114  exseis::utils::Floating_point mu =
115  (northing / k0)
116  / (eqRad
117  * (1 - std::pow(E, 2U) / 4 - 3 * std::pow(E, 4U) / 64
118  - 5 * std::pow(E, 6U) / 256));
119  exseis::utils::Floating_point phi =
120  mu + (3 * ei / 2 - 27 * std::pow(ei, 3) / 32) * std::sin(2 * mu)
121  + (21 / 16 * std::pow(ei, 2U) - 55 / 32 * std::pow(ei, 4U))
122  * std::sin(4 * mu)
123  + (151 / 96 * std::pow(ei, 3U) - 417 / 128 * std::pow(ei, 5U))
124  * std::sin(6 * mu)
125  + (1097 / 512 * std::pow(ei, 4U)) * std::sin(8 * mu);
126  exseis::utils::Floating_point c =
127  (E * E / (1 - E * E)) * std::pow(std::cos(phi), 2);
128  exseis::utils::Floating_point d =
129  x / (k0 * eqRad / std::sqrt(1 - std::pow(E * std::sin(phi), 2)));
130  lat =
131  (180 * hemi / std::acos(-1))
132  * (phi
133  - ((1 - std::pow(E * std::sin(phi), 2)) * std::tan(phi) / (1 - E * E))
134  * ((d * d / 2)
135  + ((5 + 3 * std::pow(std::tan(phi), 2) + 10 * c - 4 * c * c
136  - 9 * (E * E / (1 - E * E)))
137  * std::pow(d, 4) / 24)
138  + ((61 + 90 * std::pow(std::tan(phi), 2)
139  + 298 * std::pow(std::tan(phi), 4)
140  - 252 * (E * E / (1 - E * E)) - 3 * c * c)
141  * std::pow(d, 6) / 720)));
142  lng =
143  (6 * numZone - 183)
144  - (180
145  * (d + ((1 + 2 * std::pow(std::tan(phi), 2) + c) * std::pow(d, 3) / 6)
146  + ((5 - 2 * c + 28 * std::pow(std::tan(phi), 2) - 3 * c * c
147  + 24 * std::pow(std::tan(phi), 4) + 8 * (E * E / (1 - E * E)))
148  * std::pow(d, 5) / 120))
149  / (std::cos(phi) * std::acos(-1)));
150 }

182 /* Main function for segy to kml
