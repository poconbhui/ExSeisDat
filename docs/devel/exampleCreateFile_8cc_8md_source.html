25 void createFile(std::string name, size_t nt, size_t ns, double inc)
26 {
27  // Initialize PIOL by creating an ExSeisPIOL object
28  auto piol = ExSeis::New();
29 
30  // Create new SEGY file
31  WriteDirect file(piol, name);
32 
33  auto dec = block_decomposition(nt, piol->getNumRank(), piol->getRank());
34  size_t offset = dec.global_offset;
35  size_t lnt = dec.local_size;
36 
37  // Write some header parameters
38  file.writeNs(ns);
39  file.writeNt(nt);
40  file.writeInc(inc);
41  file.writeText("Test file\n");
42 
43  // Set and write some trace parameters
44  Param prm(lnt);
45  for (size_t j = 0; j < lnt; j++) {
46  float k = offset + j;
47  param_utils::setPrm(j, PIOL_META_xSrc, 1600.0 + k, &prm);
48  param_utils::setPrm(j, PIOL_META_ySrc, 2400.0 + k, &prm);
49  param_utils::setPrm(j, PIOL_META_xRcv, 100000.0 + k, &prm);
50  param_utils::setPrm(j, PIOL_META_yRcv, 3000000.0 + k, &prm);
51  param_utils::setPrm(j, PIOL_META_xCmp, 10000.0 + k, &prm);
52  param_utils::setPrm(j, PIOL_META_yCmp, 4000.0 + k, &prm);
53  param_utils::setPrm(j, PIOL_META_il, 2400 + k, &prm);
54  param_utils::setPrm(j, PIOL_META_xl, 1600 + k, &prm);
55  param_utils::setPrm(j, PIOL_META_tn, offset + j, &prm);
56  }
57  file.writeParam(offset, lnt, &prm);
58 
59  // Set and write some traces
60  std::vector<float> trc(lnt * ns);
61  for (size_t j = 0; j < lnt * ns; j++) {
62  trc[j] = float(offset * ns + j);
63  }
64  file.writeTrace(offset, lnt, trc.data());
65 }
