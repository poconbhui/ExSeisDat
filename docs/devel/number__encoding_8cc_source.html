17 Float_components from_IBM(
18  std::array<unsigned char, 4> ibm_float_bytes, bool big_endian)
19 {
20  // Get the IBM float in the native endian order
21  const uint32_t ibm_bits = [=]() {
22  if (big_endian == true) {
23  return (ibm_float_bytes[0] << 24) | (ibm_float_bytes[1] << 16)
24  | (ibm_float_bytes[2] << 8) | (ibm_float_bytes[3] << 0);
25  }
26 
27  return (ibm_float_bytes[3] << 24) | (ibm_float_bytes[2] << 16)
28  | (ibm_float_bytes[1] << 8) | (ibm_float_bytes[0] << 0);
29  }();

31 
32  // The IBM float is of the form
33  // SCCC CCCC QQQQ QQQQ QQQQ QQQQ QQQQ QQQQ
34  // where
35  // S is the sign bit
36  // C are the exponent bits
37  // Q are the fraction bits.
38  //
39  // The IBM number is, assuming the usual binary representation,
40  // (-1)^S * 0b0.QQQQ QQQQ QQQQ QQQQ QQQQ QQQQ * 16^(0bCCCCCC - 64)
41  //
42  // In this function, when we talk about byte "0", we mean the rightmost
43  // bit of the above representation, and byte "31" is the leftmost
44  // bit.
45 
46  // The first 24 bits make up the fraction. If it's zero we can stop,
47  // because that is defined as zero.
48  const uint32_t frac = (ibm_bits & 0x00FFFFFF);
49  if (frac == 0) {
50  return {0, 0, 0};
51  }
52 
53  // Extract the sign of the fraction (bit 31)
54  const uint32_t sign = (ibm_bits >> 31) & 0x01;
55 
56  // Extract the exponent (bits 24 to 30)
57  const uint32_t ibm_exponent_bits = (ibm_bits >> 24) & 0x7F;
58 
59  // The IBM exponent is biased by 64, and is a power of 16.
60  // We want the unbiased exponent as a power of 2.
61  // We use
62  // 16^x = 2^(4*x)
63  // to convert from base 16 to 2.
64  // We also need to use signed types to represent a signed, unbiased
65  // exponent.
66  const int32_t exp = ((int32_t(ibm_exponent_bits) - 64) * 4);
67 
68 
69  // We need to find how much we need to shift frac to have it left justified.
70  //
71  // In the IBM format, the leading 1 of the fraction is moved as far right as
72  // possible. However, since it is using a base 16 exponent, up to 3 leading
73  // bits of the fraction can be zero.
74  // We therefore, need to shift it left by 8 bits, plus those leading zeros.
75  //
76  // We use a sneaky trick here: we use the 3 rightmost bits of the IBM
77  // fraction as a lookup for a table listing the offsets. Interpreted as an
78  // integer, these have the values:
79  // 000 -> 0 ---> needs offset of 3
80  // 001 -> 1 ---> needs offset of 2
81  // 010 -> 2 ---> needs offset of 1
82  // 011 -> 3 ---> needs offset of 1
83  // and so on.
84  //
85  const uint32_t offsets[] = {3, 2, 1, 1, 0, 0, 0, 0};
86  const uint32_t leading_frac_bits = frac >> 21;
87  const int32_t shift = offsets[leading_frac_bits];
88 
89  const uint32_t shifted_frac = frac << (8 + shift);
90 
91  // We need to account for the shift in the exponent as well.
92  // We don't include the 8 bit shift, because it's not part of the IBM
93  // representation, but the extra `shift` bits are.
94  const int32_t shifted_exp = exp - shift;
95 
96  return {sign, shifted_exp, shifted_frac};
97 }

107 static uint32_t rshift_with_rounding(uint32_t value, uint32_t shift)
108 {
109 
110  // Get the part of the `value` that will be truncated. This will be the
111  // last `shift` bits
112 
113  // mask is 0000 0000 ... 1111, where the 1s are `shift` bits wide
114  const uint32_t mask = (1UL << shift) - 1;
115 
116  // trunc is the last `shift` bits of `value`
117  const uint32_t trunc = value & mask;
118 
119  // Shift and truncate `value`.
120  uint32_t truncated_value = value >> shift;
121 
122  // Apply rounding
123 
124  // We use round-half-even rounding, as that's the most common rounding
125  // implementation for IEEE numbers.
126  //
127  // In round-half-even rounding, if a number is exactly 0bxxxx.1000...,
128  // i.e. exactly half-way between two integer values, it is always
129  // rounded to the nearest even integer.
130  //
131  // The following is a list of all the potential roundings:
132  // 0bxxx0.0...1 = QQQ0
133  // 0bxxx0.10 = QQQ0
134  // 0bxxx0.1...1 = QQQ1
135  // 0bxxx1.10 = QQQ1 + 0001
136  // 0bxxx1.0...1 = QQQ1
137  // 0bxxx1.1...1 = QQQ1
138 
139 
140  // half is the value 0000...1000... representing when the truncated
141  // part is exactly half, i.e. 0b0.1000...
142  const uint32_t half = (1UL << (shift - 1));
143 
144  // If the truncated part is > 0b0.1000, round up. When it's < 0b0.1000,
145  // round down, i.e. just allow the number to be truncated.
146  if (trunc > half) {
147  truncated_value += 1;
148  }
149  else if (trunc == half) {
150  // If the truncated part is exactly half, and value is odd, round
151  // up, i.e. round to the nearest even.
152  // If it's even, just allow it to remain truncated.
153  if ((truncated_value & 0x01) == 0x01) {
154  truncated_value += 1;
155  }
156  }
157 
158  return truncated_value;
159 }

162 float to_float(Float_components components)
163 {
164  static_assert(
165  sizeof(float) == sizeof(uint32_t),
166  "to_float expects float and uint32_t to be the same size");
167 
168  uint32_t sign = components.sign;
169  int32_t exp = components.exponent;
170  uint32_t frac = components.significand;
171 
172 
173  // An IEEE number is of the form
174  // SCCC CCCC CQQQ QQQQ QQQQ QQQQ QQQQ QQQQ
175  // where
176  // S is the sign bit
177  // C are the exponent bits
178  // Q are the fraction bits.
179  //
180  // The IEEE number is, assuming the usual binary representation,
181  // (-1)^S * 0b1.QQQ QQQQ QQQQ QQQQ QQQQ QQQQ * 2^(0bCCC CCCC C - 127)
182  //
183  // We move the right-justified 1 in frac right 8 bits to position 21.
184  // It will be dropped in normal numbers, and used in denormal numbers.
185  //
186  frac = frac >> 8;
187 
188  // Now we apply the IEEE bias to the exponent.
189  // The IEEE bias is 127. We subtract an extra 1 to account for the implicit
190  // 1 in the significand.
191  exp = exp + 127 - 1;
192 
193 
194  // Denormalized numbers
195  //
196  // In IEEE, denormalized numbers are values of 2^-126 or smaller.
197  // These numbers are of the form
198  // S000 0000 0QQQ QQQQ QQQQ QQQQ QQQQ
199  // where
200  // S is the sign bit
201  // Q are the fraction bits.
202  //
203  // The denormalized IEEE number is
204  // (-1)^S * 0b0.QQQ QQQQ QQQQ QQQQ QQQQ * 2^-126
205  //
206  // Note here the initial digit is 0.QQQ, rather than 1.QQQ, as was the
207  // case with regular IEEE numbers.
208  // Converting from normal IEEE to denormalized IEEE, then, requires an
209  // extra right shift, and an explicit leading 1 in the significand.
210  //
211  // We know we need a denormalized number if our exponent is negative after
212  // biasing. In this case, the negative value represents the right shift
213  // needed to represent the number.
214  //
215  // Note: this will catch +/- 0 as well, but the frac is 0, so this shift
216  // will make no difference.
217  //
218  if (exp <= 0) {
219  // denorm numbers get an extra shift of 1, because they're in the
220  // form 0.bbbb... rather than 1.bbbb... for regular numbers.
221 
222  frac = rshift_with_rounding(frac, -exp + 1);
223  exp = 0;
224  }
225 
226  // Overflow
227  //
228  // If the exponent is larger than an IEEE representable number, we return
229  // +- inf.
230  // This number is in the form
231  // S111 1111 1000 0000 0000 0000 0000 0000
232  //
233  if (exp >= 0xFF) {
234  frac = 0;
235  exp = 0xFF;
236  }
237 
238  // Drop bit 24 of frac
239  frac = frac & 0x7FFFFF;
240 
241  // Make sure we've only got 8 bits for the exponent!
242  exp = exp & 0xFF;
243 
244  // Shift components to the positions of IEEE float.
245  // Frac is already at the correct position.
246 
247  // Sign is at bit 32
248  sign <<= 31;
249 
250  // Exp is at bits 31 to 24
251  exp <<= 23;
252 
253 
254  const uint32_t int_float = sign | exp | frac;
255 
256  float rval = 0;
257  std::memcpy(&rval, &int_float, sizeof(float));
258 
259  return rval;
260 }
