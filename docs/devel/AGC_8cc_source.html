42  for (size_t j = 0; j < win2; j++) {
43

46  // The center of the unchopped window starts at the beginning of the

58  }
59

61  // window remains unchopped.
62  for (size_t j = win2; j < signal_size - win2; j++) {
63  gain[j] =
64  gain_function(&signal[j - win2], window_size, target_amplitude, win2);
65  }
66

69

71  // the end as it approaches the boundary, down to a size of win2+1.
72  // The center of the unchopped window will always be win2 into
73  // the chopped window since we're chopping the end of the window.
74
