252  std::vector<size_t> nodups;
253  nodups.push_back(offset[idx[0]]);
254  for (size_t j = 1; j < sz; j++) {
255  if (offset[idx[j - 1]] != offset[idx[j]]) {
256  nodups.push_back(offset[idx[j]]);
257  }
258  }
259 
260  Param sprm(prm->r, (prm != PIOL_PARAM_NULL ? nodups.size() : 0LU));
261  std::vector<exseis::utils::Trace_value> strc(
262  ns * (trc != TRACE_NULL ? nodups.size() : 0LU));
263 
264  readTraceNonContiguous(
265  nodups.size(), nodups.data(), (trc != TRACE_NULL ? strc.data() : trc),
266  (prm != PIOL_PARAM_NULL ? &sprm : prm), 0LU);
267 
268  if (prm != PIOL_PARAM_NULL) {
269  for (size_t n = 0, j = 0; j < sz; ++j) {
270  if (j != 0 && offset[idx[j - 1]] != offset[idx[j]]) {
271  n++;
272  }
273 
274  param_utils::cpyPrm(n, &sprm, skip + idx[j], prm);
275  }
276  }
277 
278  if (trc != TRACE_NULL) {
279  for (size_t n = 0, j = 0; j < sz; ++j) {
280  if (j != 0 && offset[idx[j - 1]] != offset[idx[j]]) {
281  n++;
282  }
