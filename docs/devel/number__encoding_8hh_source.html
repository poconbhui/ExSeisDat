156 template<typename T>
157 std::array<unsigned char, sizeof(T)> to_big_endian(T src)
158 {
159  static_assert(
160  std::is_integral<T>::value,
161  "to_big_endian only defined for integer types.");
162 
163  std::array<unsigned char, sizeof(T)> dst;
164 
165  // For dst[i], we want the ith byte of src.
166  // We do that by shifting the ith byte into the 1st byte position and
167  // reading that first byte.
168  //
169  // Say sizeof(T) = 4.
170  // Moving the ith byte to the 1st position is equivalent to right
171  // shift by (4 - i - 1) bytes.
172  //
173  // Perhaps more obviously, we want to count from low to high in dst, and
174  // from high to low in the bytes of src.
175  //
176 
177  const size_t t_size = sizeof(T);
178 
179  for (size_t i = 0; i < t_size; i++) {
180  // Find the bitwise right shift necessary to move byte i to byte 1.
181  const size_t byte_shift = t_size - i - 1;
182  const size_t bit_shift = byte_shift * 8;
183 
184  // Read byte 1 into dst.
185  dst[i] = (src >> bit_shift) & 0xFF;
186  }
187 
188  return dst;
189 }
