235 size_t Rule::extent(void)
236 {

239  }

313 void Rule::rmRule(Meta m)
314 {
315  auto iter = translate.find(m);
316  if (iter != translate.end()) {
317 
318  RuleEntry* entry = iter->second;
319 
320  RuleEntry::MdType type = entry->type();
321  size_t num = entry->num;
322 
323  switch (type) {
324  case RuleEntry::MdType::Long:
325  numLong--;
326  break;
327 
328  case RuleEntry::MdType::Short:
329  numShort--;
330  break;
331 
332  case RuleEntry::MdType::Float:
333  numFloat--;
334  break;
335 
336  case RuleEntry::MdType::Index:
337  numIndex--;
338  break;
339 
340  case RuleEntry::MdType::Copy:
341  numCopy--;
342  break;
343  }
344 
345  delete entry;
346 
347  translate.erase(m);
348  for (auto t : translate) {
349  if (t.second->type() == type && t.second->num > num) {
350  t.second->num--;
351  }
