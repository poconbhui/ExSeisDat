434 void DataMPIIO::contigIO(
435  MFp<MPI_Status> fn,
436  size_t offset,
437  size_t sz,
438  unsigned char* d,
439  std::string msg,
440  size_t bsz,
441  size_t osz) const
442 {
443  MPI_Status stat;
444  size_t max = maxSize / osz;
445  auto vec = piol_->comm->gather<size_t>(sz);
446 
447  size_t remCall = *std::max_element(vec.begin(), vec.end());
448 
449  remCall = (remCall / max + static_cast<size_t>(remCall % max > 0))
450  - (sz / max + static_cast<size_t>(sz % max > 0));
451 
452  for (size_t i = 0; i < sz; i += max) {
453  size_t chunk = std::min(sz - i, max);
454  int err = fn(
455  file, MPI_Offset(offset + osz * i), &d[bsz * i], chunk,
456  exseis::utils::MPI_type<unsigned char>(), &stat);
457 
458  if (err != MPI_SUCCESS) {
459  log_->record(
460  name_, Logger::Layer::Data, Logger::Status::Error,
461  msg + exseis::utils::MPI_error_to_string(err, &stat),
462  PIOL_VERBOSITY_NONE);
463  }
464  }
465 
466  for (size_t i = 0; i < remCall; i++) {
467  int err =
468  fn(file, 0, NULL, 0, exseis::utils::MPI_type<unsigned char>(), &stat);
469 
470  if (err != MPI_SUCCESS) {
471  log_->record(
472  name_, Logger::Layer::Data, Logger::Status::Error,
473  msg + exseis::utils::MPI_error_to_string(err, &stat),
474  PIOL_VERBOSITY_NONE);
475  }
476  }

498  - ((sz / max) + static_cast<size_t>(sz % max > 0));

519  }
