176 unsigned char to_ASCII_from_EBCDIC(unsigned char ebcdic_char)
177 {
178  const auto compare_ebcdic = [](Character_encoding a, unsigned char b) {
179  // Search by EBCDIC
180  return a.ebcdic < b;
181  };
182 
183  // Search for `ebcdic_char` in `ebcdic_sorted_array`
184  const auto& ebcdic_char_it = std::lower_bound(
185  std::begin(ebcdic_sorted_array), std::end(ebcdic_sorted_array),
186  ebcdic_char, compare_ebcdic);
187 
188  if (ebcdic_char_it == std::end(ebcdic_sorted_array)) {
189  return char_sub.ascii;
190  }
191  return ebcdic_char_it->ascii;
192 }
