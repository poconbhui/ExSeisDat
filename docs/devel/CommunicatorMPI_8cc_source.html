59 struct MPIManager {
62  MPIManager()
63  {
64  // If we're not managing MPI, just do nothing.
65  if (managingMPI() == ManagingMPI::no) {
66  return;
67  }
68 
69  int initialized = 0;
70  MPI_Initialized(&initialized);
71 
72  if (initialized == 0) {
73  MPI_Init(NULL, NULL);
74  }
75 
76  // Set managingMPI value if the user hasn't already
77  if (managingMPI() == ManagingMPI::unset) {
78  if (initialized != 0) {
79  // MPI was already initialized
80  managingMPI() = ManagingMPI::no;
81  }
82  else {
83  // We initialized MPI
84  managingMPI() = ManagingMPI::yes;
85  }
86  }
87  }
