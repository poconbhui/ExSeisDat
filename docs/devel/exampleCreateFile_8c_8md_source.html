17 void createFile(const char* name, size_t nt, size_t ns, size_t inc)
18 {
19  // Initialise the PIOL by creating an ExSeis object
20  PIOL_ExSeis* piol = PIOL_ExSeis_new(PIOL_VERBOSITY_NONE);
21 
22  // Create a SEGY file object
23  PIOL_File_WriteDirect* fh = PIOL_File_WriteDirect_new(piol, name);
24 
25  // Perform some decomposition (user decides how they will decompose)
26  struct exseis_Contiguous_decomposition dec = exseis_block_decomposition(
27  nt, PIOL_ExSeis_getNumRank(piol), PIOL_ExSeis_getRank(piol));
28 
29  // The offset for the local process
30  size_t offset = dec.global_offset;
31  // The number of traces for the local process to handle
32  size_t lnt = dec.local_size;
33 
34  // Write some header parameters
35  PIOL_File_WriteDirect_writeNs(fh, ns);
36  PIOL_File_WriteDirect_writeNt(fh, nt);
37  PIOL_File_WriteDirect_writeInc(fh, inc);
38  PIOL_File_WriteDirect_writeText(fh, "Test file\n");
39 
40  // Set some trace parameters
41  PIOL_File_Param* prm = PIOL_File_Param_new(NULL, lnt);
42  for (size_t j = 0; j < lnt; j++) {
43  float k = offset + j;
44  PIOL_File_setPrm_double(j, PIOL_META_xSrc, 1600.0 + k, prm);
45  PIOL_File_setPrm_double(j, PIOL_META_ySrc, 2400.0 + k, prm);
46  PIOL_File_setPrm_double(j, PIOL_META_xRcv, 100000.0 + k, prm);
47  PIOL_File_setPrm_double(j, PIOL_META_yRcv, 3000000.0 + k, prm);
48  PIOL_File_setPrm_double(j, PIOL_META_xCmp, 10000.0 + k, prm);
49  PIOL_File_setPrm_double(j, PIOL_META_yCmp, 4000.0 + k, prm);
50  PIOL_File_setPrm_Integer(j, PIOL_META_il, 2400 + offset + j, prm);
51  PIOL_File_setPrm_Integer(j, PIOL_META_xl, 1600 + offset + j, prm);
52  PIOL_File_setPrm_Integer(j, PIOL_META_tn, offset + j, prm);
53  }
54 
55  // Set some traces
56  assert(lnt * ns > 0);
57  float* trc = calloc(lnt * ns, sizeof(float));
58  for (size_t j = 0; j < lnt * ns; j++) {
59  trc[j] = (float)(offset * ns + j);
60  }
61 
62  // Write the traces and trace parameters
63  PIOL_File_WriteDirect_writeTrace(fh, offset, lnt, trc, prm);
64 
65  // Free the data
66  PIOL_File_Param_delete(prm);
67  free(trc);
68 
69  // Close the file handle and free the piol
70  PIOL_File_WriteDirect_delete(fh);
71  PIOL_ExSeis_delete(piol);
72 }
