221 int16_t find_scalar(exseis::utils::Floating_point val)
222 {
223  constexpr exseis::utils::Integer tenk = 10000LL;
224 
225  // First we need to determine what scale is required to store the
226  // biggest decimal value of the int.
227  exseis::utils::Integer integer_part = exseis::utils::Integer(val);
228  int32_t int_part = integer_part;
229 
230  if (integer_part != int_part) {
231  // Starting with the smallest scale factor, see what is the smallest
232  // scale we can apply and still hold the integer portion.
233  // We drop as much precision as it takes to store the most significant
234  // digit.
235  for (int32_t scal = 10; scal <= tenk; scal *= 10) {
236  exseis::utils::Integer v = integer_part / scal;
237  int32_t iv = v;
238 
239  if (v == iv) {
240  return scal;
241  }
242  }
243 
244  return 0;
245  }
246  else {
247  // Get the first four digits
248  exseis::utils::Integer digits =
249  std::llround(val * exseis::utils::Floating_point(tenk))
250  - integer_part * tenk;
251  // if the digits are all zero we don't need any scaling
252  if (digits != 0) {
253  // We try the most negative scale values we can first.
254  //(scale = - 10000 / i)
255  for (int32_t i = 1; i < tenk; i *= 10) {
256  if ((digits % (i * 10)) != 0) {
257  int16_t scaleFactor = -tenk / i;
258  // Now we test that we can still store the most significant
259  // byte
260  exseis::utils::Floating_point scal =
261  parse_scalar(scaleFactor);
262 
263  // int32_t t = exseis::utils::Integer(val / scal) - digits;
264  int32_t t = std::lround(val / scal);
265  t /= -scaleFactor;
266 
267  if (t == integer_part) {
268  return scaleFactor;
269  }
270  }
271  }
272  }
273  return 1;
274  }
