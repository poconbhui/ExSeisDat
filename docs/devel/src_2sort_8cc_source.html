572 template<class T>
573 void sort(
574  ExSeisPIOL* piol,
575  size_t regionSz,
576  std::vector<T>& temp1,
577  std::vector<T>& dat,
578  Compare<T> comp = nullptr)
579 {
580  size_t lnt = dat.size();
581  size_t numRank = piol->comm->getNumRank();
582  size_t rank = piol->comm->getRank();
583  size_t edge1 = (rank ? regionSz : 0LU);
584  size_t edge2 = (rank != numRank - 1LU ? regionSz : 0LU);
585 
586  if (comp != nullptr) {
587  std::sort(
588  dat.begin(), dat.begin() + lnt,
589  [comp](auto& a, auto& b) -> bool { return comp(a, b); });
590  }
591  else {
592  std::sort(dat.begin(), dat.begin() + lnt);
593  }
594 
595  std::copy(dat.begin(), dat.end(), temp1.begin());
596  std::vector<T> temp2;
597  // Infinite loop if there is more than one process, otherwise no loop
598  while (numRank > 1) {
599  temp2 = temp1;
600  sendLeft(piol, regionSz, temp1);
601 
602  if (comp != nullptr) {
603  std::sort(
604  temp1.begin() + edge1, temp1.end(),
605  [comp](auto& a, auto& b) -> bool { return comp(a, b); });
606  }
607  else {
608  // default pair sorting is first then second
609  std::sort(temp1.begin() + edge1, temp1.end());
610  }
611 
612  sendRight(piol, regionSz, temp1);
613 
614  if (comp != nullptr) {
615  std::sort(
616  temp1.begin(), temp1.end() - edge2,
617  [comp](auto& a, auto& b) -> bool { return comp(a, b); });
618  }
619  else {
620  std::sort(temp1.begin(), temp1.end() - edge2);
621  }
622 
623  int reduced = 0;
624  for (size_t j = 0; j < lnt && !reduced; j++) {
625  reduced += (temp1[j] != temp2[j]);
626  }
627  int greduced = 1;
628 
629  int err = MPI_Allreduce(
630  &reduced, &greduced, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
631 
632  if (err != MPI_SUCCESS) {
633  piol->log->record(
634  "", Logger::Layer::Ops, Logger::Status::Error,
635  "Sort MPI_Allreduce error: "s
636  + exseis::utils::MPI_error_to_string(err),
637  PIOL_VERBOSITY_NONE);
638  }
639 
640  if (!greduced) {
641  break;
642  }
643  }
644 
645  for (size_t i = 0; i < lnt; i++) {
646  dat[i] = temp1[i];
647  }
648 }

655 std::vector<size_t> sort(ExSeisPIOL* piol, std::vector<size_t> list)
656 {
657  const size_t lnt = list.size();
658  const size_t regionSz = piol->comm->min(lnt) / 4LU;
659  const size_t edge2 =
660  (piol->comm->getRank() != piol->comm->getNumRank() - 1 ? regionSz : 0LU);
661  const size_t offset = piol->comm->offset(lnt);
662 
663  std::vector<std::pair<size_t, size_t>> plist(lnt);
664  for (size_t i = 0; i < lnt; i++) {
665  plist[i].first = list[i];
666  plist[i].second = offset + i;
667  }
668 
669  {

671 
672 // Work-around for bug in the intel compiler (intel/2016-u3) on Fionn with
673 // std::pair less-than operator
674 #if defined(__INTEL_COMPILER)
675  Compare<std::pair<size_t, size_t>> check =
676  [](

678  const std::pair<size_t, size_t>& e2) -> bool {
679  return (
680  e1.first < e2.first

868 void sortP(ExSeisPIOL* piol, Param* prm, CompareP comp = nullptr)
869 {
870  size_t lnt = prm->size();
871  size_t numRank = piol->comm->getNumRank();
872  size_t rank = piol->comm->getRank();
873  size_t regionSz = piol->comm->min(lnt) / 4LU;
874  size_t edge1 = (rank != 0 ? regionSz : 0LU);
875  size_t edge2 = (rank != numRank - 1 ? regionSz : 0LU);
876 
877  std::vector<size_t> t1(lnt);
878  std::iota(t1.begin(), t1.end(), 0LU);
879  std::sort(t1.begin(), t1.end(), [prm, comp](size_t& a, size_t& b) -> bool {
880  return comp(prm, a, b);
881  });
882 
883  Param temp1(prm->r, lnt + edge2);
884  Param temp2(temp1.r, temp1.size());
885  Param temp3(prm->r, temp1.size());
886 
887  for (size_t i = 0; i < lnt; i++) {
888  param_utils::cpyPrm(t1[i], prm, i, &temp1);
889  }
890 
891  // Infinite loop if there is more than one process, otherwise no loop
892  while (numRank > 1) {
893  temp2 = temp1;
894  sendLeft(piol, regionSz, &temp1);
895 
896  {
897  std::vector<size_t> t1(temp1.size() - edge1);
898  std::iota(t1.begin(), t1.end(), edge1);
899  std::sort(
900  t1.begin(), t1.end(),
901  [&temp1, comp](size_t& a, size_t& b) -> bool {
902  return comp(&temp1, a, b);
903  });
904 
905  for (size_t i = 0; i < t1.size(); i++) {
906  param_utils::cpyPrm(t1[i], &temp1, i, &temp3);
907  }
908 
909  for (size_t i = 0; i < t1.size(); i++) {
910  param_utils::cpyPrm(i, &temp3, i + edge1, &temp1);
911  }
912  }
913 
914  sendRight(piol, regionSz, &temp1);
915 
916  {
917  std::vector<size_t> t1(temp1.size() - edge2);
918  std::iota(t1.begin(), t1.end(), 0);
919  std::sort(
920  t1.begin(), t1.end(),
921  [&temp1, comp](size_t& a, size_t& b) -> bool {
922  return comp(&temp1, a, b);
923  });
924 
925  for (size_t i = 0; i < t1.size(); i++) {
926  param_utils::cpyPrm(t1[i], &temp1, i, &temp3);
927  }
928 
929  for (size_t i = 0; i < t1.size(); i++) {
930  param_utils::cpyPrm(i, &temp3, i, &temp1);
931  }
932  }
933 
934  int reduced = 0;
935  for (size_t j = 0; j < lnt && reduced == 0; j++) {
936  const auto param1 = param_utils::getPrm<exseis::utils::Integer>(
937  j, PIOL_META_gtn, &temp1);
938 
939  const auto param2 = param_utils::getPrm<exseis::utils::Integer>(
940  j, PIOL_META_gtn, &temp2);
941 
942  if (param1 != param2) {
943  reduced++;
944  }
945  }
946  int greduced = 1;
947 
948  int err = MPI_Allreduce(
949  &reduced, &greduced, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
950 
951  if (err != MPI_SUCCESS) {
952  piol->log->record(
953  "", Logger::Layer::Ops, Logger::Status::Error,
954  "Sort MPI_Allreduce error: "
955  + exseis::utils::MPI_error_to_string(err),
956  PIOL_VERBOSITY_NONE);
957  }
958 
959  if (greduced == 0) {
960  break;
961  }
962  }
963 
964  for (size_t i = 0; i < lnt; i++) {
965  param_utils::cpyPrm(i, &temp1, i, prm);
966  }
967 }
