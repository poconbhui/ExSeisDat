48 Trace_value triangular_RMS_gain(
49  const Trace_value* signal,
50  size_t window_size,
51  Trace_value target_amplitude,
52  size_t window_center)
53 {
54  assert(window_size > 0);
55  assert(window_size > window_center);
56 
57  // When the window center is near the start or the end of the range of
58  // signals, the window can go outside this range.
59  //
60  // Say we have the following setup:
61  //
62  // signals: [****************]
63  // window: [-------]
64  // window center: *
65  // scaling: ^
66  // / \
67  // / \
68  // / \
69  //
70  // We treat the array of signals as though it's padded with zeros:
71  //
72  // signals: [00****************]
73  // window: [-------]
74  // window center: X
75  // scaling: ^
76  // / \
77  // / \
78  // / \
79  //
80  // But the view we pass into this function is:
81  //
82  // signals: [****************]
83  // window: [-----]
84  // window center: ^
85  // scaling: / \
86  // \
87  // \
88  //
89  // That is, the signals are passed in from as low a value as possible,
90  // the window size is set to the number of addressable elements in the
91  // signal, and the window center is set to some non-central position.
92  //
93  // So, we can't rely on the window center to actually be at the center,
94  // but we know the center to *one* of the ends will be half the width,
95  // and in particular, whichever is bigger will be the half-width.
96  //
97 
98  // 0 to window_center
99  const size_t left_width = window_center;
100 
101  // window_center to window_size - 1.
102  const size_t right_width = (window_size - 1) - window_center;
103 
104  const Trace_value half_width = std::max(left_width, right_width);
105 
106  Trace_value amp = 0;
107  for (size_t j = 0; j < window_size; j++) {
108  // A signed-sensitive std::abs(j - window_center)
109  const auto distance_from_center =
110  (j > window_center) ? j - window_center : window_center - j;
111 
112  // Linear interpolation:
113  // 0 at distance_from_center == half_width
114  // 1 at distance_from_center == 0
115  const auto scaling = 1 - distance_from_center / half_width;
116 
117  const auto scaled_signal = signal[j] * scaling;
118 
119  amp += scaled_signal * scaled_signal;
120  }
121  assert(amp != 0);
122 
123  const auto non_zero = [](Trace_value i) { return i != 0; };
124  const size_t num = std::count_if(signal, &signal[window_size], non_zero);
125 
126  return target_amplitude / std::sqrt(amp / std::max<size_t>(1, num));
127 }
