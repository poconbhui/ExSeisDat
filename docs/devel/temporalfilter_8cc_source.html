121  Complex_trace_value* z,

151  size_t N,

183 void makeFilter(
184  FltrType type,
185  exseis::utils::Trace_value* numer,
186  exseis::utils::Trace_value* denom,
187  exseis::utils::Integer N,
188  exseis::utils::Trace_value fs,
189  exseis::utils::Trace_value cf1,
190  exseis::utils::Trace_value cf2)
191 {
192  size_t tN = (cf2 == 0) ? N : N * 2LU;
193  exseis::utils::Trace_value Wn =
194  4 * std::tan(pi * (cf1 / (fs * exseis::utils::Trace_value(0.5))) / 2);
195  exseis::utils::Trace_value W2 =
196  4 * std::tan(pi * (cf2 / (fs * exseis::utils::Trace_value(0.5))) / 2);
197 
198  std::vector<Complex_trace_value> z(tN);
199  std::vector<Complex_trace_value> p(tN);
200 
201  // Determines the analogue zero-pole-gain representation of filter for given
202  // number of poles
203  for (exseis::utils::Integer i = 0; i < N; i++) {
204  p[i] = -exp(
205  I * pi * exseis::utils::Trace_value(1LL + 2LL * i - N)
206  / exseis::utils::Trace_value(2LL * N));
207  }
208 
209  // Determines the analogue zero-pole-gain for given filter type at specific
210  // frequency(ies)
211  exseis::utils::Trace_value k;
212  switch (type) {
213  default:
214  case FltrType::Lowpass:
215  k = lowpass(N, z.data(), p.data(), Wn);
216  break;
217 
218  case FltrType::Highpass:
219  k = highpass(N, z.data(), p.data(), Wn);
220  break;
221 
222  case FltrType::Bandpass:
223  k = bandpass(N, z.data(), p.data(), Wn, W2);
224  break;
225 
226  case FltrType::Bandstop:
227  k = bandstop(N, z.data(), p.data(), Wn, W2);
228  break;
229  }
230 
231  expandPoly(z.data(), tN, numer);
232  expandPoly(p.data(), tN, denom);
233 
234  for (size_t i = 0LU; i < tN + 1LU; i++) {
235  numer[i] *= k;
236  }
237 
238  for (size_t i = 0LU; i < (tN + 1LU) / 2LU; i++) {
239  std::swap(denom[i], denom[tN - i]);
