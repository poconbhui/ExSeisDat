203 fourd_t dsr(
204  const fourd_t xs1,
205  const fourd_t ys1,
206  const fourd_t xr1,
207  const fourd_t yr1,
208  const fourd_t xs2,
209  const fourd_t ys2,
210  const fourd_t xr2,
211  const fourd_t yr2)
212 {
213  // Forward-boat case
214  const fourd_t forward =
215  hypot(xs1 - xs2, ys1 - ys2) + hypot(xr1 - xr2, yr1 - yr2);
216  // Reverse-boat case
217  const fourd_t reverse =
218  hypot(xs1 - xr2, ys1 - yr2) + hypot(xr1 - xs2, yr1 - ys2);
219  return std::min(forward, reverse);
220 }

285  // of src x.

287  rstart++) {
288  }
289  for (; rend >= rstart && crd2->xSrc[rend] > crd1->xSrc[lend - 1] + dsrmax;
290  rend--) {
291  }
292  for (; lstart < lend && crd1->xSrc[lstart] < crd2->xSrc[rstart] - dsrmax;
293  lstart++) {
294  }

296  lend--) {
297  }
298 
299  // TODO: Check if theoretical speedup is realisable for this alignment
300  lstart = size_t(lstart / ALIGN) * ALIGN;
301  size_t lsz = lend - lstart;
302  size_t rsz = rend - rstart;
303 
304  // Copy min and minrs to aligned memory
305  fourd_t* lminrs;
306  size_t* lmin;
307  posix_memalign(
308  reinterpret_cast<void**>(&lminrs), ALIGN, crd1->sz * sizeof(fourd_t));
309  posix_memalign(
310  reinterpret_cast<void**>(&lmin), ALIGN, crd1->sz * sizeof(size_t));
311  std::copy(min.begin(), min.begin() + crd1->sz, lmin);
312  std::copy(minrs.begin(), minrs.begin() + crd1->sz, lminrs);
313 
314  // These declarations are so that the compiler handles the pragma correctly
315  const fourd_t* xS1 = crd1->xSrc;
316  const fourd_t* xR1 = crd1->xRcv;
317  const fourd_t* yS1 = crd1->ySrc;
318  const fourd_t* yR1 = crd1->yRcv;
319  const fourd_t* xS2 = crd2->xSrc;
320  const fourd_t* xR2 = crd2->xRcv;
321  const fourd_t* yS2 = crd2->ySrc;
322  const fourd_t* yR2 = crd2->yRcv;

324 
325  // Loop through every file1 trace
326  #pragma omp simd aligned(xS2:ALIGN) aligned(yS2:ALIGN) aligned(xR2:ALIGN) \
327  aligned(yR2:ALIGN) aligned(xS1:ALIGN) aligned(yS1:ALIGN) \
328  aligned(xR1:ALIGN) aligned(yR1:ALIGN) aligned(tn:ALIGN) \

330  for (size_t i = lstart; i < lend; i++) {
331  const fourd_t xs1 = xS1[i], ys1 = yS1[i], xr1 = xR1[i], yr1 = yR1[i];
332  size_t lm = lmin[i];
333  fourd_t lmrs = lminrs[i];
334  // loop through a multiple of the alignment
335  for (size_t j = rstart; j < rend; j++) {
336  const fourd_t xs2 = xS2[j], ys2 = yS2[j], xr2 = xR2[j],
337  yr2 = yR2[j];

339  (!ixline
340  || (crd1->il[i] == crd2->il[j]
341  && crd1->xl[i] == crd2->xl[j]) ?
342  dsr(xs1, ys1, xr1, yr1, xs2, ys2, xr2, yr2) :
343  std::numeric_limits<fourd_t>::max());
344  // Update min if applicable
345  lm = (dval < lmrs ? tn[j] : lm);
346  // Update minrs if applicable
347  lmrs = std::min(dval, lmrs);
348  }
349  lmin[i] = lm;
350  lminrs[i] = lmrs;
351  }
352 
353  std::copy(lmin, lmin + crd1->sz, min.begin());
354  std::copy(lminrs, lminrs + crd1->sz, minrs.begin());
355  free(lmin);
356  free(lminrs);
357  return lsz * rsz;
358 }

440 void calc4DBin(
441  ExSeisPIOL* piol,
442  const fourd_t dsrmax,
443  const Coords* crd1,
444  const Coords* crd2,
445  const FourDOpt opt,
446  std::vector<size_t>& min,
447  std::vector<fourd_t>& minrs)
448 {
449  cmsg(piol, "Compute phase");
450  size_t rank = piol->comm->getRank();
451  size_t numRank = piol->comm->getNumRank();
452  auto szall = piol->comm->gather(std::vector<size_t>{crd2->sz});
453 
454  // The File2 min/max from every process
455  auto xsmin = piol->comm->gather(std::vector<fourd_t>{crd2->xSrc[0LU]});
456  auto xsmax =
457  piol->comm->gather(std::vector<fourd_t>{crd2->xSrc[crd2->sz - 1LU]});
458 
459  // The File1 local min and local maximum for the particular process
460  auto xslmin = crd1->xSrc[0LU];
461  auto xslmax = crd1->xSrc[crd1->sz - 1LU];
462 
463  // Perform a local initialisation update of min and minrs
464  if (opt.ixline) {
465  initUpdate<true>(crd1, crd2, min, minrs);
466  }
467  else {
468  initUpdate<false>(crd1, crd2, min, minrs);
469  }
470 
471  // This for loop determines the processes the local process will need to be
472  // communicating with.
473  std::vector<size_t> active;
474  for (size_t i = 0LU; i < numRank; i++) {
475  if ((xsmin[i] - dsrmax <= xslmax) && (xsmax[i] + dsrmax >= xslmin)) {
476  active.push_back(i);
477  }
478  }
479 
480  if (opt.verbose) {
481  printxSMinMax(piol, xslmin, xslmax, xsmin[rank], xsmax[rank], active);
482  printxSrcMinMax(piol, xsmin, xsmax);
483  }
484 
485  auto time = MPI_Wtime();
486 #define ONE_WAY_COMM
487 #ifdef ONE_WAY_COMM
488  auto win = createCoordsWin(crd2, opt.ixline);
489 #else
490  std::vector<MPI_Request> reqs;
491  {
492  auto xsfmin = piol->comm->gather<fourd_t>(xslmin);
493  auto xsfmax = piol->comm->gather<fourd_t>(xslmax);
494 
495  for (size_t i = 0LU; i < numRank; i++)
496  if (
497  (xsmin[rank] - dsrmax <= xsfmax[i])
498  && (xsmax[rank] + dsrmax >= xsfmin[i])) {
499  std::vector<MPI_Request> rq = sendCrd(i, crd1, opt.ixline);
500  reqs.insert(reqs.end(), rq.begin(), rq.end());
501  }
502  }
503  piol->comm->barrier();
504 #endif
505  // Load balancing would make sense since the workloads are non-symmetric.
506  // It might be difficult to do load-balancing though considering workload
507  // very much depends on the constraints from both processess
508 
509  // Perform the updates of min and minrs using data from other processes.
510  // This is the main loop.
511  for (size_t i = 0; i < active.size(); i++) {
512  double ltime = MPI_Wtime();
513  size_t lrank = active[i];
514 #ifdef ONE_WAY_COMM
515  auto proc = getCoordsWin(lrank, szall[lrank], win, opt.ixline);
516 #else
517  auto proc = recvCrd(lrank, szall[lrank], opt.ixline);
518 #endif
519  double wtime = MPI_Wtime() - ltime;
520  double sent = szall[lrank]
521  * (4LU * sizeof(fourd_t) + sizeof(size_t)
522  + 2LU * sizeof(exseis::utils::Integer));
523  size_t ops =
524  (opt.ixline ? update<true>(crd1, proc.get(), min, minrs, dsrmax) :
525  update<false>(crd1, proc.get(), min, minrs, dsrmax));
526 
527  std::cout << rank << "\t-->\t" << active[i]
528  << "\tTime: " << MPI_Wtime() - ltime - wtime << " sec."
529  << " Comm: " << wtime << " sec "
530  << sent / wtime / pow(2., 30.) << " GB/s. OpSave: "
531  << (double(crd1->sz * proc->sz - ops))
532  / double(crd1->sz * proc->sz) * 100.0
533  << std::endl;
534  }
535 
536  std::cout << rank << " done. "
537  << "Total rounds: " << active.size()
538  << " Time: " << MPI_Wtime() - time << " seconds" << std::endl;
539 
540 #ifdef ONE_WAY_COMM
541  for (size_t i = 0; i < win.size(); i++) {
542  MPIErr(MPI_Win_free(&win[i]));
543  }
544 #else
545 
546  std::vector<MPI_Status> stat(reqs.size());
547  MPIErr(MPI_Waitall(reqs.size(), reqs.data(), stat.data()));
548 
549 #endif
550  piol->comm->barrier();
551  cmsg(
552  piol, "Compute phase completed in " + std::to_string(MPI_Wtime() - time)
553  + " seconds");
554 }
