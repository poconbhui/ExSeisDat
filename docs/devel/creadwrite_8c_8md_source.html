36 void readWriteFullTrace(
37  PIOL_ExSeis* piol,
38  PIOL_File_ReadDirect* ifh,
39  PIOL_File_WriteDirect* ofh,
40  size_t off,
41  size_t tcnt,
42  ModTrc ftrc,
43  ModPrm fprm)
44 {
45  PIOL_File_Param* trhdr = PIOL_File_Param_new(NULL, tcnt);
46  size_t ns = PIOL_File_ReadDirect_readNs(ifh);
47 
48  assert(tcnt * PIOL_SEGY_utils_getDFSz(ns) > 0);
49  float* trace = malloc(tcnt * PIOL_SEGY_utils_getDFSz(ns));
50  assert(trace);
51 
52  PIOL_File_ReadDirect_readTrace(ifh, off, tcnt, trace, trhdr);
53  if (ftrc != NULL) {
54  for (size_t i = 0; i < tcnt; i++) {
55  ftrc(off, ns, &trace[i]);
56  }
57  }
58 
59  if (fprm != NULL) {
60  for (size_t i = 0; i < tcnt; i++) {
61  fprm(off, i, trhdr);
62  }
63  }
64 
65  PIOL_File_WriteDirect_writeTrace(ofh, off, tcnt, trace, trhdr);
66  PIOL_ExSeis_isErr(piol, NULL);
67 
68  free(trace);
69  PIOL_File_Param_delete(trhdr);
70 }

72 void writePayload(
73  PIOL_ExSeis* piol,
74  PIOL_File_ReadDirect* ifh,
75  PIOL_File_WriteDirect* ofh,
76  size_t goff,
77  size_t lnt,
78  size_t tcnt,
79  ModPrm fprm,
80  ModTrc ftrc)
81 {
82  size_t biggest = lnt;
83  MPI_Allreduce(
84  &lnt, &biggest, 1, MPI_UNSIGNED_LONG, MPI_MAX, MPI_COMM_WORLD);
85  size_t extra =
86  biggest / tcnt - lnt / tcnt + (biggest % tcnt > 0) - (lnt % tcnt > 0);
87 
88  for (size_t i = 0U; i < lnt; i += tcnt) {
89  readWriteFullTrace(
90  piol, ifh, ofh, goff + i, (i + tcnt < lnt ? tcnt : lnt - i), ftrc,
91  fprm);
92  }
93 
94  for (size_t i = 0U; i < extra; i++) {
95  readWriteFullTrace(piol, ifh, ofh, goff, 0, ftrc, fprm);
96  }
97 }

150 int main(int argc, char** argv)
151 {
152  // Flags:
153  // -i input file
154  // -o output file
155  // -m maximum memory
156  // -p modify trace paramaters
157  // -t modify trace values
158  char* opt = "i:o:m:pt"; // TODO: uses a GNU extension
159 
160  char* iname = NULL;
161  char* oname = NULL;
162 
163  size_t memmax = 2U * 1024U * 1024U * 1024U; // bytes
164 
165  ModPrm modPrm = NULL;
166  ModTrc modTrc = false;
167 
168  for (int c = getopt(argc, argv, opt); c != -1;
169  c = getopt(argc, argv, opt)) {
170 
171  const size_t optarg_length = strlen(optarg) + 1;
172 
173  switch (c) {
174  case 'i':
175  // TODO: POSIX is vague about the lifetime of optarg. Next
176  // function may be unnecessary
177  free(iname);
178  iname = malloc(optarg_length * sizeof(char));
179 
180  strncpy(iname, optarg, optarg_length);
181 
182  break;
183 
184  case 'o':
185  free(oname);
186  oname = malloc(optarg_length * sizeof(char));
187 
188  strncpy(oname, optarg, optarg_length);
189 
190  break;
191 
192  case 'm':
193  if (sscanf(optarg, "%zu", &memmax) != 1) {
194  fprintf(stderr, "Incorrect arguments to memmax option\n");
195 
196  free(iname);
197  free(oname);
198 
199  return -1;
200  }
201 
202  break;
203 
204  case 'p':
205  printf("The trace parameters will be modified\n");
206  modPrm = SourceX1600Y2400;
207 
208  break;
209 
210  case 't':
211  printf("The traces will be modified\n");
212  modTrc = TraceLinearInc;
213 
214  break;
215 
216  default:
217  fprintf(
218  stderr, "One of the command line arguments is invalid\n");
219 
220  break;
221  }
222  }
223  assert(iname && oname);
224 
225  PIOL_ExSeis* piol = PIOL_ExSeis_new(PIOL_VERBOSITY_NONE);
226  PIOL_ExSeis_isErr(piol, NULL);
227 
228  ReadWriteFile(piol, iname, oname, memmax, modPrm, modTrc);
229 
230  PIOL_ExSeis_delete(piol);
231 
232  free(iname);
233  free(oname);
234 
235  return 0;
236 }
