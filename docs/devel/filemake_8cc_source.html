149  size_t offset = 0;

170  size_t extra = biggest / max + (biggest % max > 0 ? 1 : 0)
171  - (lnt / max + (lnt % max > 0 ? 1 : 0));
172  if (random) {
173  writeRandom(*piol, &file, nt, ns, lnt, extra, max);
174  }
